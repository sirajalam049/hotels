---- Database Schema ----

Hotel Booking

  1. hotels
    1. id ----- > int(11) | AUTO_INCREMENT | PRIMARY_KEY
    2. name ----- > varchar(150)
    3. phone ----- > varchar(15)
    4. location ----- > varchar(150)
    5. address ----- > varchar(256)
    6. price ----- > smallint(15)
    7. keywords ----- > text
    8. visitors ----- > int(11)
            
            CREATE TABLE hotels(
                id int(11) NOT NULL AUTO_INCREMENT, 
                name varchar(150) NOT NULL, 
                phone varchar(15) NOT NULL, 
                total_rooms smallint(3) NOT NULL DEFAULT 1, 
                location varchar(150) NOT NULL, 
                address varchar(256) NOT NULL, 
                price varchar(15) NOT NULL, 
                keywords text NOT NULL, 
                visitors int(11) DEFAULT 1, 
                PRIMARY KEY(id) 
            )
  2. users
    1. id ----- > int(11) NOT NULL, AUTO_INCREMENT
    2. name ----- > varchar(128) NOT NULL
    3. email ----- > varchar(256) NOT NULL
    4. pswd ----- > varchar(512) NOT NULL
            
            CREATE TABLE users(
                id int(11) NOT NULL AUTO_INCREMENT,
                name varchar(128) NOT NULL,
                email varchar(256) NOT NULL UNIQUE,
                pswd varchar(512) NOT NULL,
                PRIMARY KEY(id)
            )
  3. bookings
    1. id ----- > int(11) NOT NULL AUTO_INCREMENT
    2. date ----- > DATE
    3. hotel_id ----- > int(11) NOT NULL
    4. user_id ----- > int(11) NOT NULL
    5. total_rooms ----- > smallint(3) NOT NULL DEFAULT 1
    6. days ----- > int(3) NOT NULL DEFAULT 1
    7. booking_date ----- > DATETIME
            
            CREATE TABLE drafts(
                id int(11) NOT NULL AUTO_INCREMENT,
                date DATE NOT NULL,
                hotel_id int(11) NOT NULL,
                user_id int(11) NOT NULL,
                total_rooms smallint(3) NOT NULL DEFAULT 1,
                days int(3) NOT NULL DEFAULT 1,
                booking_date DATETIME NOT NULL,
                PRIMARY KEY(id),
                FOREIGN KEY(hotel_id) REFERENCES hotels(id)
                    ON DELETE CASCADE
                    ON UPDATE CASCADE,
                FOREIGN KEY(user_id) REFERENCES users(id)
                    ON DELETE CASCADE
                    ON UPDATE CASCADE
            )
  4. drafts
    1. id ----- > int(11) NOT NULL AUTO_INCREMENT
    2. date ----- > DATETIME
    3. hotel_id ----- > int(11)
    4. user_id ----- > int(11)
    5. total_rooms ----- > smallint(3)
    6. days ----- > int(3)
            
            CREATE TABLE drafts(
                id int(11) NOT NULL AUTO_INCREMENT,
                date DATE,
                hotel_id int(11),
                user_id int(11) NOT NULL,
                total_rooms smallint(3),
                days int(3),
                PRIMARY KEY(id),
                FOREIGN KEY(hotel_id) REFERENCES hotels(id)
                    ON DELETE CASCADE
                    ON UPDATE CASCADE,
                FOREIGN KEY(user_id) REFERENCES users(id)
                    ON DELETE CASCADE
                    ON UPDATE CASCADE
            )
